 20 function is_point_in_poly(poly, _x, _y){
 21     for(var c = false, i = -1, l = poly.length, j = l - 1; ++i < l; j = i)
 22         ((poly[i].y <= _y && _y < poly[j].y) || (poly[j].y <= _y && _y < poly[i].y))
 23         && (_x < (poly[j].x - poly[i].x) * (_y - poly[i].y) / (poly[j].y - poly[i].y) + poly[i].x)
 24         && (c = !c);
 25     return c;
 26 }

 28 // use poly_area(nodes,true) for signed area
 29 function poly_area(nodes) {
 30 	var area = 0
 31 	nodes.each(function(node,index) {
 32 		if (index < nodes.length-1) next = nodes[index+1]
 33 		else next = nodes[0]
 34 		if (index > 0) last = nodes[index-1]
 35 		else last = nodes[nodes.length-1]
 36 		area += last.x*node.y-node.x*last.y+node.x*next.y-next.x*node.y
 37 	})
 38 	if (arguments[1] == true) return area/2
 39 	else return Math.abs(area/2)
 40 }

 57 			cx += (polygon[i].x + polygon[j].x) * factor
 58 			cy += (polygon[i].y + polygon[j].y) * factor
 59 		}
 60 		
 61 		a *= 6
 62 		factor = 1/a
 63 		cx *= factor
 64 		cy *= factor
 65 		centroid[0] = cx
 66 		centroid[1] = cy
 67 		return centroid
