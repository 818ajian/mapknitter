 14 function mousemove(event) { 
 15 	Mouse.x = -1*Event.pointerX(event)
 16 	Mouse.y = -1*Event.pointerY(event)
 17 	draw()
 18 }
 19 
 20 function wheel(event){
 21 	var delta = 0;
 22 	if (!event) event = window.event;
 23 	if (event.wheelDelta) {
 24 		delta = event.wheelDelta/120;
 25 		if (window.opera) delta = -delta;
 26 	} else if (event.detail) {
 27 		delta = -event.detail/3;
 28 	}
 29 	if (delta && !Cartagen.live_gss) {
 30 		if (delta <0) {
 31 			Cartagen.zoom_level += delta/40
 32 		} else {
 33 			Cartagen.zoom_level += delta/40
 34 		}
 35 		if (Cartagen.zoom_level < Cartagen.zoom_out_limit) Cartagen.zoom_level = Cartagen.zoom_out_limit
 36 	}
 37 	draw()
 38 }

 40 function events_init() {
 41 	// Observe mouse events:
 42 	body = $('body')
 43 	var canvas_el = $('canvas')
 44 	canvas_el.observe('mousemove', mousemove)
 45 	canvas_el.observe('mousedown', mousedown)
 46 	canvas_el.observe('mouseup', mouseup)
 47 	canvas_el.observe('dblclick', doubleclick)
 48 	// Observe scrollwheel:
 49 	if (window.addEventListener) window.addEventListener('DOMMouseScroll', wheel, false)
 50 	window.onmousewheel = document.onmousewheel = wheel
 51 	
 52 	Event.observe(document, 'keypress', function(e) {
 53 		var code;
 54 		if (!e) var e = window.event;
 55 		if (e.keyCode) code = e.keyCode;
 56 		else if (e.which) code = e.which;
 57 		var character = String.fromCharCode(code);
 58 		if (key_input) {
 59 			if (character == "s") zoom_in()
 60 			if (character == "w") zoom_out()
 61 			if (character == "d") Map.rotate += 0.1
 62 			if (character == "a") Map.rotate -= 0.1
 63 			if (character == "f") Map.x += 20/Cartagen.zoom_level
 64 			if (character == "h") Map.x -= 20/Cartagen.zoom_level
 65 			if (character == "t") Map.y += 20/Cartagen.zoom_level
 66 			if (character == "g") Map.y -= 20/Cartagen.zoom_level
 67 		} else {
 68 			// just modifiers:
 69 			switch(character){
 70 				case "r": keys.set("r",true)
 71 				break
 72 				case "z": keys.set("z",true)
 73 				break
 74 				case "g": 
 75 					if (!Cartagen.live_gss) Cartagen.show_gss_editor()
 76 				break
 77 				case "h": get_static_plot('/static/rome/highway.js')
 78 			}
 79 		}
 80 		draw()
 81 	});
 82 	Event.observe(document, 'keyup', function() {
 83 		keys.set("r",false)
 84 		keys.set("z",false)
 85 		switch (single_key) {
 86 		}
 87 		single_key = null
 88 	});
 89 	
 90 	// iPhone events:
 91 	if (Prototype.Browser.MobileSafari || window.PhoneGap) {
 92 		// get rid of url bar:
 93 		
 94 		// addEventListener("load", function() { setTimeout(updateLayout, 0) }, false)
 95 		// var currentWidth = 0;
 96 		// function updateLayout() {
 97 		//     if (window.innerWidth != currentWidth) {
 98 		//         currentWidth = window.innerWidth;
 99 		//         var orient = currentWidth == 320 ? "profile" : "landscape";
100 		//         document.body.setAttribute("orient", orient);
101 		//         setTimeout(function() {
102 		//             window.scrollTo(0, 1);
103 		//         }, 100);           
104 		//     }
105 		// }
106 		// setInterval(updateLayout, 400);
107 	
108 		canvas_el.ontouchstart = function(e){
109 			e.preventDefault();
110 			if(e.touches.length == 1){ // Only deal with one finger
111 		 		var touch = e.touches[0]; // Get the information for finger #1
112 			    var node = touch.target; // Find the node the drag started from
113 	
114 				mouseDown = true
115 				clickFrame = frame
116 				Mouse.click_x = touch.screenX
117 				Mouse.click_y = touch.screenY
118 				Map.x_old = Map.x
119 				Map.y_old = Map.y
120 				if (!dragging) {
121 					globalDragging = true
122 				}
123 				draw()	
124 			  }
125 		}
126 		canvas_el.ontouchmove = function(e) {	
127 			e.preventDefault();
128 			if(e.touches.length == 1){ // Only deal with one finger
129 				var touch = e.touches[0]; // Get the information for finger #1
130 				var node = touch.target; // Find the node the drag started from
131 	
132 				drag_x = (touch.screenX - Mouse.click_x)
133 				drag_y = (touch.screenY - Mouse.click_y)
134 	
135 				var d_x = Math.cos(Map.rotate)*drag_x+Math.sin(Map.rotate)*drag_y
136 				var d_y = Math.cos(Map.rotate)*drag_y-Math.sin(Map.rotate)*drag_x
137 	
138 				Map.x = Map.x_old+(d_x/Cartagen.zoom_level)
139 				Map.y = Map.y_old+(d_y/Cartagen.zoom_level)
140 
141 				draw()
142 			}
143 		}
144 		canvas_el.ontouchend = function(e) {
145 			if(e.touches.length == 1) {
146 				mouseUp = true
147 				mouseDown = false
148 				releaseFrame = frame
149 				globalDragging = false
150 				dragging = false
151 			}
152 			User.update()
153 			draw()
154 		}
155 		canvas_el.ongesturestart = function(e) {
156 			zoom_level_old = Cartagen.zoom_level
157 		}
158 		canvas_el.ongesturechange = function(e){
159 		  var node = e.target;
160 			if (Map.rotate_old == null) Map.rotate_old = Map.rotate
161 			Map.rotate = Map.rotate_old + (e.rotation/180)*Math.PI
162 			Cartagen.zoom_level = zoom_level_old*e.scale
163 			draw()
164 		}
165 		canvas_el.ongestureend = function(e){
166 			Map.rotate_old = null
167 			User.update()
168 		}	
169 	}
170 }

172 function doubleclick(event) {
173 	on_object = false
174 	objects.each(function(object) { 
175 		if (!on_object && overlaps(object.x,object.y,Mouse.x,Mouse.y,0)) {
176 			object.doubleclick()
177 			on_object = true
178 		}
179 	})
180 }

182 function drag() {
183 	if (globalDragging && !Prototype.Browser.MobileSafari && !window.PhoneGap) {
184 		drag_x = (Mouse.x - Mouse.click_x)
185 		drag_y = (Mouse.y - Mouse.click_y)
186 		if (keys.get("r")) { // rotating
187 			Map.rotate = Map.rotate_old + (-1*drag_y/height)
188 		} else if (keys.get("z")) {
189 			if (Cartagen.zoom_level > 0) {
190 				Cartagen.zoom_level = Math.abs(Cartagen.zoom_level - (drag_y/height))
191 			} else {
192 				Cartagen.zoom_level = 0
193 			}
194 		} else {
195 			// var h = Math.sqrt((drag_x*drag_x)+(drag_y*drag_y))
196 			var d_x = Math.cos(Map.rotate)*drag_x+Math.sin(Map.rotate)*drag_y
197 			var d_y = Math.cos(Map.rotate)*drag_y-Math.sin(Map.rotate)*drag_x
198 			
199 			Map.x = Map.x_old+(d_x/Cartagen.zoom_level)
200 			Map.y = Map.y_old+(d_y/Cartagen.zoom_level)
201 			// Map.x = Map.x_old+(drag_x/Cartagen.zoom_level)
202 			// Map.y = Map.y_old+(drag_y/Cartagen.zoom_level)
203 		}
204 	}
205 }

207 function mousedown(event) {
208 	mouseDown = true
209 	clickFrame = frame
210 	Mouse.click_x = Mouse.x
211 	Mouse.click_y = Mouse.y
212 	Map.x_old = Map.x
213 	Map.y_old = Map.y
214 	Map.rotate_old = Map.rotate
215 	if (!dragging) {
216 		globalDragging = true
217 	}
218 }
219 
220 function mouseup() {
221 	mouseUp = true
222 	mouseDown = false
223 	releaseFrame = frame
224 	globalDragging = false
225 	dragging = false
226 	User.update()
227 }
228 
229 function clickLength() {
230 	return releaseFrame-clickFrame
231 }
