1 /**
  2  * @namespace
  3  * Contains methods and variables for spacially indexing features using
  4  * geohashes.
  5  */
  6 var Geohash = {
  7 	hash: new Hash(),
  8 	objects: [],
  9 	grid: false,
 10 	default_length: 6, // default length of geohash
 11 	limit_bottom: 8, // 12 is most ever...
 12 	// once-per-frame calls to regenerate objects, etc.
 13 	draw: function() {
 14 		this.get_objects()
 15 	},
 16 	// adds a feature to a geohash index
 17 	put: function(lat,lon,feature,length) {
 18 		if (!length) length = this.default_length
 19 		var key = Geohash.get_key(lat,lon,length)
 20 		// check to see if the geohash is already populated:
 21 		var merge_hash = this.hash.get(key)
 22 		if (!merge_hash) {
 23 			merge_hash = [feature]
 24 		} else {
 25 			merge_hash.push(feature)
 26 		}
 27 		this.hash.set(key,merge_hash)
 28 	},
 29 	put_object: function(object) {
 30 		Geohash.put(Projection.y_to_lat(object.y),Projection.x_to_lon(object.x),object,Geohash.get_key_length(object.width,object.height)-1)
 31 	},
 32 	get_key: function(lat,lon,length) {
 33 		if (!length) length = this.default_length
 34 		if (length < 1) length = 1
 35 		return encodeGeoHash(lat,lon).truncate(length,"")
 36 	},
 37 	// fetch features in a geohash from lat/lon and length
 38 	get: function(lat,lon,length) {
 39 		if (!length) length = this.default_length
 40 		var key = this.get_key(lat,lon,length)
 41 		return this.hash.get(key)
 42 	},
 43 	// fetch features in a geohash key
 44 	get_from_key: function(key) {
 45 		// this.draw_bbox(key)
 46 		var result = this.hash.get(key)
 47 		if (result) return result
 48 		else return []
 49 	},
 50 	// fetch features in a geohash from a geohash key, and all shorter keys
 51 	get_upward: function(key) {
 52 		if (key.length > this.limit_bottom) key.truncate(this.limit_bottom,'')
 53 		// this.draw_bbox(key)
 54 		var this_level = this.hash.get(key)
 55 		if (this_level && key.length > 0) {
 56 			// Cartagen.debug(key+': '+this_level.length)
 57 			if (key.length > 1) return this_level.concat(this.get_upward(key.truncate(key.length-1,"")))
 58 			else return this_level
 59 		} else {
 60 			// Cartagen.debug(key+': 0')
 61 			if (key.length > 1) return this.get_upward(key.truncate(key.length-1,""))
 62 			else return []
 63 		}
 64 	},
 65 	/** 
 66 	 *  Geohash.get_keys_upward(key,keys) -> undefined
 67 	 *  
 68 	 *  fetch keys in a geohash from a geohash key, and all shorter keys
 69 	 *  checks for redundancy against a hash of keys 
 70 	 **/
 71 	get_keys_upward: function(key) {
 72 		if (key.length > this.limit_bottom) key.truncate(this.limit_bottom,'')
 73 		if (key.length > 0) {
 74 			// this.draw_bbox(key)
 75 			Geohash.keys.set(key,true)
 76 			k = key.truncate(key.length-1,"")
 77 			if (key.length > 1 && !Geohash.keys.get(k)) {
 78 				this.get_keys_upward(k)
 79 			}
 80 		}
 81 	},
 82 	// fetch adjacent geohashes:
 83 	get_neighbors: function(key) {
 84 		var neighbors = []
 85 		var dirs = ['top','bottom','left','right']
 86 		dirs.each(function(dir) {
 87 			var n_key = calculateAdjacent(key,dir)
 88 			var n_array = this.get_from_key(n_key)
 89 			if (n_array) neighbors = neighbors.concat(n_array)
 90 			// Cartagen.debug("n_key: "+n_key)
 91 		},this)
 92 		// Cartagen.debug('neighbors of '+key+': '+neighbors.length)
 93 		return neighbors
 94 	},
 95 	// return a geohash length from a zoom_level
 96 	// i believe this is deprecated (jeff)
 97 	length_from_zoom: function(zoom_level) {
 98 		return zoom_level/0.003
 99 	},
100 	/** 
101 	 *  Geohash.fill_bbox(key,keys) -> undefined
102 	 *  
103 	 *  given a geohash key, recurses outwards to neighbors while still within the viewport
104 	 **/
105 	fill_bbox: function(key,keys) {
106 		var dirs = ['top','bottom','left','right']
107 		// we may be able to improve efficiency by only checking certain directions
108 		dirs.each(function(dir) {
109 			var k = calculateAdjacent(key,dir)
110 			if (!keys.get(k)) {
111 				keys.set(k,true)
112 				// if still inside viewport:
113 				var bbox = decodeGeoHash(k) //[lon1, lat2, lon2, lat1]
114 				if (in_range(bbox.latitude[2],Map.bbox[3],Map.bbox[1]) && in_range(bbox.longitude[2],Map.bbox[0],Map.bbox[2])) this.fill_bbox(k,keys)
115 				// if (overlaps(bbox.latitude[2],bbox.longitude[2],Map.lat,Map.lon,Math.min(Map.lat_height,Map.lon_width)/2)) this.fill_bbox(k,keys)
116 				this.draw_bbox(k)
117 			}
118 		},this)
119 	},
120 	trace: function() {
121 		this.hash.keys().each(function(key) {
122 			Cartagen.debug(key+': '+this.hash.get(key).length)
123 		},this)
124 		return this.hash.size()
125 	},
126 	bbox: function(geohash) {
127 		var geo = decodeGeoHash(geohash)
128 		return [geo.longitude[0],geo.latitude[1],geo.longitude[1],geo.latitude[0]]
129 	},
130 	draw_bbox: function(key) {
131 		if (Geohash.grid) {
132 			var bbox = this.bbox(key)
133 			canvas.lineWidth = 1/Cartagen.zoom_level
134 			$C.stroke_style('rgba(0,0,0,0.5)')
135 			// Cartagen.debug(key.length+": "+(bbox[2]-bbox[0])+","+(bbox[1]-bbox[3]))
136 			var width = (Projection.lon_to_x(bbox[2])-Projection.lon_to_x(bbox[0]))
137 			var height = (Projection.lat_to_y(bbox[1])-Projection.lat_to_y(bbox[3]))
138 			$C.stroke_rect(-width-Projection.lon_to_x(bbox[0]),Projection.lat_to_y(bbox[3]),width,height)
139 			$C.fill_style('rgba(0,0,0,0.5)')
140 			canvas.font = (9/Cartagen.zoom_level)+"pt Helvetica"
141 			canvas.fillText(key,-width-Projection.lon_to_x(bbox[0])+3/Cartagen.zoom_level,Projection.lat_to_y(bbox[3])-3/Cartagen.zoom_level)
142 			// Cartagen.debug(key+": xy_rect: "+Projection.lon_to_x(bbox[0])+","+Projection.lat_to_y(bbox[3])+","+(Projection.lon_to_x(bbox[2])-Projection.lon_to_x(bbox[0]))+","+(Projection.lat_to_y(bbox[1])-Projection.lat_to_y(bbox[3])))
143 			// Cartagen.debug(key+': latlon_rect: '+bbox[0]+','+bbox[3]+','+bbox[2]+','+bbox[1])
144 		}
145 	},
146 	get_key_length: function(lat,lon) {
147 		if (lon < 0.0000003357) lon_key = 12
148 		else if (lon < 0.000001341) lon_key = 11
149 		else if (lon < 0.00001072) lon_key = 10
150 		else if (lon < 0.00004291) lon_key = 9
151 		else if (lon < 0.0003433) lon_key = 8
152 		else if (lon < 0.001373) lon_key = 7
153 		else if (lon < 0.01098) lon_key = 6
154 		else if (lon < 0.04394) lon_key = 5
155 		else if (lon < 0.3515) lon_key = 4
156 		else if (lon < 1.406) lon_key = 3
157 		else if (lon < 11.25) lon_key = 2
158 		else if (lon < 45) lon_key = 1
159 		else lon_key = 0 // eventually we can map the whole planet at once
160 		
161 		if (lat < 0.0000001676) lat_key = 12
162 		else if (lat < 0.000001341) lat_key = 11
163 		else if (lat < 0.000005364) lat_key = 10
164 		else if (lat < 0.00004291) lat_key = 9
165 		else if (lat < 0.0001716) lat_key = 8
166 		else if (lat < 0.001373) lat_key = 7
167 		else if (lat < 0.005493) lat_key = 6
168 		else if (lat < 0.04394) lat_key = 5
169 		else if (lat < 0.1757) lat_key = 4
170 		else if (lat < 1.40625) lat_key = 3
171 		else if (lat < 5.625) lat_key = 2
172 		else if (lat < 45) lat_key = 1
173 		else lat_key = 0 // eventually we can map the whole planet at once
174 		return Math.min(lat_key,lon_key)
175 	},
176 	get_objects: function() {
177 		this.objects = []
178 
179 		// get geohash for each of the 4 corners,
180 		this.keys = new Hash
181 		this.key_length = Geohash.get_key_length(0.0015/Cartagen.zoom_level,0.0015/Cartagen.zoom_level)
182 		
183 		this.key = Geohash.get_key(Map.lat,Map.lon,this.key_length)
184 		
185 		var bbox = decodeGeoHash(this.key) //[lon1, lat2, lon2, lat1]
186 		
187 		this.fill_bbox(this.key,this.keys)
188 		this.get_keys_upward(this.key)
189 
190 		this.keys.keys().each(function(key,index) {
191 			this.get_keys_upward(key)
192 		},this)
193 		
194 		this.keys.keys().each(function(key) {
195 			this.objects = this.objects.concat(this.get_from_key(key))
196 		},this)
197 		
198 		// reverse because smaller objects are added first:
199 		return this.objects.reverse
200 	}
