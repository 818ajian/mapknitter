  4 var Style = {
  5 	styles: {
  6 		// this doesn't get used. We don't have a body object to load them into...

 13 	style_body: function() {
 14 		if (Style.styles) {
 15 			if (Style.styles.body.fillStyle) $C.fill_style(Style.styles.body.fillStyle)
 16 			if (Style.styles.body.strokeStyle) $C.stroke_style(Style.styles.body.strokeStyle)
 17 			if (Style.styles.body.lineWidth || Style.styles.body.lineWidth == 0) $C.line_width(Style.styles.body.lineWidth)
 18 			if (Style.styles.body.pattern && Object.isUndefined(Style.styles.body.pattern_img)) {
 19 				Style.styles.body.pattern_img = new Image()
 20 				Style.styles.body.pattern_img.src = Style.styles.body.pattern
 21 			}
 22 			if (Style.styles.body.pattern_img) {
 23 				try {
 24 					$C.fill_style(canvas.createPattern(Style.styles.body.pattern_img,'repeat'))
 25 				} catch(e) {}
 26 			}
 27 			$C.rect(0,0,width,height)
 28 			$C.stroke_rect(0,0,width,height)
 29 		}
 30 		canvas.lineJoin = 'round'
 31 		canvas.lineCap = 'round'
 32 	},
 33 	parse_styles: function(feature,selector) {
 34 		try {
 35 			// check for function or parameter for each style type...
 36 			// or is it copying the function itself, and doesn't need to know if it's a function or parameter?
 37 			if (selector.opacity) feature.opacity = selector.opacity
 38 			if (selector.fillStyle) feature.fillStyle = selector.fillStyle
 39 			if (selector.lineWidth || selector.lineWidth == 0) feature.lineWidth = selector.lineWidth
 40 			if (selector.strokeStyle && Object.isFunction(selector.strokeStyle)) {
 41 				// bind the styles object to the context of this Way:
 42 				feature.strokeStyle = selector.$C.stroke_style()
 43 			} else {
 44 				feature.strokeStyle = selector.strokeStyle
 45 			}
 46 			// patterns
 47 			if (selector.pattern) {
 48 				feature.pattern_img = new Image()
 49 				feature.pattern_img.src = selector.pattern
 50 			}
 51 			// radius is relevant to nodes, i.e. single points
 52 			if (selector.radius) feature.radius = selector.radius
 53 			// check selector for hover:
 54 			if (selector['hover']) feature.hover = selector['hover']
 55 			if (selector['mouseDown']) feature.mouseDown = selector['mouseDown']
 56 
 57 			// copy styles based on feature name
 58 			if (Style.styles[feature.name] && Style.styles[feature.name].fillStyle) feature.fillStyle = Style.styles[feature.name].fillStyle
 59 			if (Style.styles[feature.name] && Style.styles[feature.name].strokeStyle) feature.strokeStyle = Style.styles[feature.name].strokeStyle
 60 
 61 			// font styling:
 62 			if (selector.fontColor) feature.label.fontColor = selector.fontColor
 63 			if (selector.fontSize) feature.label.fontSize = selector.fontSize
 64 			if (selector.fontScale) feature.label.fontScale = selector.fontScale
 65 			if (selector.fontBackground) feature.label.fontBackground = selector.fontBackground
 66 			if (selector.text) feature.label.text = selector.text
 67 
 68 			feature.tags.each(function(tag) {
 69 				//look for a style like this:
 70 				if (Style.styles[tag.key]) {
 71 					if (Style.styles[tag.key].opacity) feature.opacity = Style.styles[tag.key].opacity
 72 					if (Style.styles[tag.key].fillStyle) feature.fillStyle = Style.styles[tag.key].fillStyle
 73 					if (Style.styles[tag.key].strokeStyle) feature.strokeStyle = Style.styles[tag.key].strokeStyle
 74 					if (Style.styles[tag.key].lineWidth) feature.lineWidth = Style.styles[tag.key].lineWidth
 75 					if (Style.styles[tag.key].fontColor) feature.label.fontColor = Style.styles[tag.key].fontColor
 76 					if (Style.styles[tag.key].fontSize) feature.label.fontSize = Style.styles[tag.key].fontSize
 77 					if (Style.styles[tag.key].fontScale) feature.label.fontScale = Style.styles[tag.key].fontScale
 78 					if (Style.styles[tag.key].fontBackground) feature.label.fontBackground = Style.styles[tag.key].fontBackground
 79 					if (Style.styles[tag.key].text) {
 80 						if (Object.isFunction(Style.styles[tag.key].text)) feature.label.text = Style.styles[tag.key].text.apply(feature)
 81 						else feature.label.text = Style.styles[tag.key].text
 82 					}
 83 					if (Style.styles[tag.key].pattern) {
 84 						feature.pattern_img = new Image()
 85 						feature.pattern_img.src = Style.styles[tag.key].pattern
 86 					}
 87 				}
 88 				if (Style.styles[tag.value]) {
 89 					if (Style.styles[tag.value].opacity) feature.opacity = Style.styles[tag.value].opacity
 90 					if (Style.styles[tag.value].fillStyle) feature.fillStyle = Style.styles[tag.value].fillStyle
 91 					if (Style.styles[tag.value].strokeStyle) feature.strokeStyle = Style.styles[tag.value].strokeStyle
 92 					if (Style.styles[tag.value].lineWidth) feature.label.lineWidth = Style.styles[tag.value].lineWidth
 93 					if (Style.styles[tag.value].fontColor) feature.label.fontColor = Style.styles[tag.value].fontColor
 94 					if (Style.styles[tag.value].fontSize) feature.label.fontSize = Style.styles[tag.value].fontSize
 95 					if (Style.styles[tag.value].fontScale) feature.label.fontScale = Style.styles[tag.value].fontScale
 96 					if (Style.styles[tag.value].fontBackground) feature.label.fontBackground = Style.styles[tag.value].fontBackground
 97 					if (Style.styles[tag.value].text) {
 98 						if (Object.isFunction(Style.styles[tag.value].text)) feature.label.text = Style.styles[tag.value].text.apply(feature)
 99 						else feature.label.text = Style.styles[tag.value].text
100 					}
101 					if (Style.styles[tag.value].pattern) {
102 						feature.pattern_img = new Image()
103 						feature.pattern_img.src = Style.styles[tag.value].pattern
104 					}
105 				}
106 
107 				//check tags for hover:
108 				if (Style.styles[tag.key] && Style.styles[tag.key]['hover']) {
109 					feature.hover = Style.styles[tag.key]['hover']
110 				}
111 				if (Style.styles[tag.value] && Style.styles[tag.value]['hover']) {
112 					feature.hover = Style.styles[tag.value]['hover']
113 				}
114 				//check tags for mouseDown:
115 				if (Style.styles[tag.key] && Style.styles[tag.key]['mouseDown']) {
116 					feature.mouseDown = Style.styles[tag.key]['mouseDown']
117 				}
118 				if (Style.styles[tag.value] && Style.styles[tag.value]['mouseDown']) {
119 					feature.mouseDown = Style.styles[tag.value]['mouseDown']
120 				}
121 				// check tags for refresh:
122 				if (Style.styles[tag.key] && Style.styles[tag.key]['refresh']) {
123 
124 					$H(Style.styles[tag.key]['refresh']).each(function(pair) {
125 						Style.create_refresher(feature, pair.key, pair.value)
126 					})
127 				}
128 				if (Style.styles[tag.value] && Style.styles[tag.value]['refresh']) {
129 					if(!feature.style_generators) feature.style_generators = {}
130 					$H(Style.styles[tag.value]['refresh']).each(function(pair) {
131 						Style.create_refresher(feature, pair.key, pair.value)
132 					})
133 				}
134 			})
135 		} catch(e) {
136 			Cartagen.debug("There was an error in your stylesheet. Please check http://wiki.cartagen.org for the GSS spec. Error: "+trace(e))
137 		}
138 	},
139 	create_refresher: function(feature, property, interval) {
140 		if (Object.isFunction(feature[property])) { //sanity check
141             if (['fontBackground', 'fontColor', 'fontScale', 'fontSize', 'text'].include(property)) {
142                 feature = feature.label
143             }
144 			if(!feature.style_generators) feature.style_generators = {}
145 			if(!feature.style_generators.executers) feature.style_generators.executers = {}
146 			feature.style_generators[property] = feature[property]
147 			Style.refresh_style(feature, property)
148 			feature.style_generators.executers[property] = new PeriodicalExecuter(function() {
149 				Style.refresh_style(feature, property)
150 			}, interval)
151 		}
152 	},
153 	refresh_style: function(feature, property) {
154 		feature[property] = feature.style_generators[property]()
155 	},
156 	// sets the canvas 'pen' styles using the object.foo style definitions
157 	apply_style: function(feature) {
158 		canvas.globalOpacity = 1
159 		if (feature.opacity) {
160 			canvas.globalOpacity = Object.value(feature.opacity)
161 		}
162 		if (feature.strokeStyle) {
163 			 $C.stroke_style(Object.value(feature.strokeStyle))
164 		}
165 		if (feature.fillStyle) {
166 			$C.fill_style(Object.value(feature.fillStyle))
167 		}
168 		if (feature.pattern_img) {
169 			$C.fill_style(canvas.createPattern(feature.pattern_img,'repeat'))
170 		}
171 		if (feature.lineWidth) {
172 			$C.line_width(Object.value(feature.lineWidth))
173 		}
174 		
175 		// trigger hover and mouseDown styles:
176 		if (feature instanceof Way) {
177 			if (feature.hover && feature.closed_poly && is_point_in_poly(feature.nodes,Map.pointer_x(),Map.pointer_y())) {
178 				Style.apply_style(feature.hover)
179 				if (!Object.isUndefined(feature.hover.action)) feature.hover.action()
180 			}
181 			if (feature.mouseDown && mouseDown == true && feature.closed_poly && is_point_in_poly(feature.nodes,Map.pointer_x(),Map.pointer_y())) {
182 				Style.apply_style(feature.mouseDown)
183 				if (!Object.isUndefined(feature.mouseDown.action)) feature.mouseDown.action()
184 			}
185 		} else if (feature instanceof Node) {
186 			if (feature.hover && overlaps(feature.x,feature.y,Map.pointer_x(),Map.pointer_y(),100)) {
187 				Style.apply_style(feature.hover)
188 				if (feature.hover.action) feature.hover.action()
189 			}
190 			if (feature.mouseDown && mouseDown == true && overlaps(feature.x,feature.y,Map.pointer_x(),Map.pointer_y(),100)) {
191 				Style.apply_style(feature.mouseDown)
192 				if (feature.mouseDown.action) feature.mouseDown.action()
193 			}
194 		}
195 	},
196 	// same as apply_style but just for fonts. This was necessary because
197 	// strokeStyle and such have to be reset *after* drawing actual polygons but
198 	// *before* drawing text.
199 	apply_font_style: function(feature) {
200 		if (feature.fontColor) {
201 			if (Object.isFunction(feature.fontColor)) $C.stroke_style(feature.fontColor())
202 			else $C.stroke_style(feature.fontColor)
203 		}

210 	// load a remote stylesheet, given a URL
211 	load_styles: function(stylesheet_url) {
212 		if (stylesheet_url[0,4] == "http") {
213 			stylesheet_url = "/map/style?url="+stylesheet_url
214 		}
215 		new Ajax.Request(stylesheet_url,{
216 			method: 'get',
217 			onComplete: function(result) {
218 				Cartagen.debug('applying '+stylesheet_url)
219 				Style.styles = ("{"+result.responseText+"}").evalJSON()
220 				Style.stylesheet_source = "{"+result.responseText+"}"
221 				Style.apply_gss(Style.stylesheet_source)
222 				// populate the gss field
223 				if($('gss_textarea')) {
224 					$('gss_textarea').value = Style.stylesheet_source
225 				}
226 			}
227 		})
228 	},
229 	// given a string of gss, applies the string to all Ways and Nodes in the objects array
230 	apply_gss: function(gss) {
231 		if (Object.isUndefined(arguments[1])) var clear_styles = true
232 		else clear_styles = arguments[1]
233 		Style.styles = gss.evalJSON()
234 		objects.each(function(object) {
235 			if (clear_styles) {
236 				object.lineWeight = null
237 				object.strokeStyle = null
238 				object.fillStyle = null
239 				object.hover = null
240 				object.mouseDown = null
241 			}
242 			if (object instanceof Node) Style.parse_styles(object,Style.styles.node)
243 			if (object instanceof Way) Style.parse_styles(object,Style.styles.way)
244 		},this)
245 	}
