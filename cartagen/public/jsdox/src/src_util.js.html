 21 function strstr( haystack, needle, bool ) {
 22     var pos = 0;
 23 
 24     haystack += '';
 25     pos = haystack.indexOf( needle );
 26     if (pos == -1) {
 27         return false;
 28     } else{
 29         if( bool ){
 30             return haystack.substr( 0, pos );
 31         } else{
 32             return haystack.slice( pos );
 33         }
 34     }
 35 }
