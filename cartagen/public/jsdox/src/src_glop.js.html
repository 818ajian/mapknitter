 14 function draw() {
 15 	$C.clear()
 16 	if (Cartagen.fullscreen) {
 17 		width = document.viewport.getWidth()
 18 		height = document.viewport.getHeight()
 19 		$('canvas').width = width
 20 		$('canvas').height = height
 21 		$$('body')[0].style.width = width+"px"
 22 	}
 23 	else {
 24 		width = $('canvas').getWidth()
 25 		height = $('canvas').getHeight()
 26 		$('canvas').width = width
 27 		$('canvas').height = height
 28 	}
 29 	frame += 1
 30 	try { drag() } catch(e) {}
 31 
 32 	// let additional script subscribe to the draw method:
 33 	draw_calls.each(function(call) {
 34 		
 35 	})
 36 	// cartagen-specific calls
 37 	if (typeof Cartagen != "undefined") Cartagen.draw()
 38 	else {
 39 		objects.each(function(object) { 
 40 			object.draw()
 41 		})
 42 	}
 43 
 44     // cartagen-specific call - draws labels so that labels are drawn after
 45     // other objects.
 46 	if (typeof Cartagen != "undefined") Cartagen.post_draw()
 47 	
 48 	if (mouseDown) {
 49 		mouseDown = false
 50 	}
 51 	if (mouseUp) {
 52 		mouseUp = false
 53 	}
 54 }

 80 function jsonify(input,newlines) {
 81 	if (newlines == null) var newline = ""
 82 	else var newline = "\r"
 83 	var json = ""
 84 	if (input instanceof Array) {
 85 		var string = ''
 86 		input.each(function(item) {
 87 			string += jsonify(item)+","+newline
 88 		})
 89 		string = string.truncate(string.length-1,'')
 90 		json += "["+string+"]"
 91 	} else if (Object.isString(input)) {
 92 		json += "'"+String(input).escapeHTML()+"'"
 93 	} else if (Object.isNumber(input)) {
 94 		json += String(input)
 95 	} else if (typeof input == 'object') {
 96 		var string = ''
 97 		Object.keys(input).each(function(key,index) {
 98 			string += key+": "+jsonify(Object.values(input)[index])+", "+newline
 99 		})
100 		string.truncate(string.length-1)
101 		json += "{"+string+"}"
102 	} else {
103 		json += String(input).escapeHTML()
104 	}
105 	return json
106 }

108 function deep_clone(obj) {
109     var c = {};
110     for (var i in obj) {
111         var prop = obj[i];
112  
113         if (prop instanceof Array) {
114 			c[i] = prop.slice();
115         } else if (typeof prop == 'object') {
116            c[i] = deep_clone(prop);
117 		} else {
118            c[i] = prop;
119         }
120     }
121     return c;
122 }
