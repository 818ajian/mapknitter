  2 var Way = Class.create({

  4 	highlight: false,

  7 	closed_poly: false,

 12     initialize: function(data) {
 13 		Object.extend(this, data)
 14 		if (this.nodes.length > 1 && this.nodes[0].x == this.nodes[this.nodes.length-1].x && this.nodes[0].y == this.nodes[this.nodes.length-1].y) this.closed_poly = true
 15 		if (this.tags.get('natural') == "coastline") this.closed_poly = true
 16 		if (this.closed_poly) {
 17 			var centroid = Geometry.poly_centroid(this.nodes)
 18 			this.x = centroid[0]*2
 19 			this.y = centroid[1]*2
 20 		} else {
 21 		// attempt to make letters follow line segments:
 22 			this.x = (this.middle_segment()[0].x+this.middle_segment()[1].x)/2
 23 			this.y = (this.middle_segment()[0].y+this.middle_segment()[1].y)/2
 24 		}
 25 		this.area = poly_area(this.nodes)
 26 		this.label = new Label(this)
 27 		this.bbox = Geometry.calculate_bounding_box(this.nodes)
 28 			// calculate longest dimension to file in a correct geohash:
 29 			this.width = Math.abs(Projection.x_to_lon(this.bbox[1])-Projection.x_to_lon(this.bbox[3]))
 30 			this.height = Math.abs(Projection.y_to_lat(this.bbox[0])-Projection.y_to_lat(this.bbox[2]))
 31 		Style.parse_styles(this,Style.styles.way)
 32 		objects.push(this) // made obsolete by Geohash
 33 		Geohash.put_object(this)
 34 		Cartagen.ways.set(this.id,this)
 35     },
 36 	// returns the middle-most line segment as a tuple [node_1,node_2]
 37 	middle_segment: function() {
 38 		// Cartagen.debug(this.nodes[Math.floor(this.nodes.length/2)+1])
 39         if (this.nodes.length == 1) {
 40             return [this.nodes[0], this.nodes[0]]
 41         }
 42         else if (this.nodes.length == 2) {
 43             return [this.nodes[0], this.nodes[1]]
 44         }
 45         else {
 46             return [this.nodes[Math.floor(this.nodes.length/2)],this.nodes[Math.floor(this.nodes.length/2)+1]]
 47         }
 48 	},
 49 	middle_segment_angle: function() {
 50         var segment = this.middle_segment()
 51         if (segment[1]) {
 52             var _x = segment[0].x-segment[1].x
 53             var _y = segment[0].y-segment[1].y
 54             return (Math.tan(_y/_x)/1.7)
 55         } else return 90
 56 	},
 57 	draw: function() {
 58 		Cartagen.object_count++
 59 		// only draw if in the viewport:
 60 		if (intersect(Viewport.bbox[0],Viewport.bbox[1],Viewport.bbox[2],Viewport.bbox[3],this.bbox[0],this.bbox[1],this.bbox[2],this.bbox[3])) {
 61 			Cartagen.way_count++
 62 			this.shape()
 63 			this.age += 1;
 64 		}
 65 	},
 66 	shape: function() {

 68 			Style.apply_style(this)
 69 			if (this.highlight) {

 71 				$C.stroke_style("red")
 72 			}
 73 			// fade in after load:
 74 			if (Object.isUndefined(this.opacity)) this.opacity = 1
 75 			if (this.age < 20) {
 76 				canvas.globalAlpha = this.opacity*(this.age/20)
 77 			} else {
 78 				canvas.globalAlpha = this.opacity
 79 			}
 80 
 81 		$C.begin_path()
 82 		$C.move_to(this.nodes[0].x,this.nodes[0].y)
 83 
 84 		if (Map.resolution == 0) Map.resolution = 1
 85 		this.nodes.each(function(node,index){
 86 			if ((index % Map.resolution == 0) || index == 0 || index == this.nodes.length-1 || this.nodes.length <= 30) {
 87 				Cartagen.node_count++
 88 				$C.line_to(node.x,node.y)
 89 			}
 90 		},this)
 91 
 92 		// fill the polygon if the beginning and end nodes are the same.
 93 		// we'll have to change this for open polys, like coastlines
 94 		$C.stroke()
 95 		if (this.closed_poly) $C.fill()
 96 
 97 		// show bboxes for ways:

 99 		// $C.stroke_style('red')
100 		// $C.stroke_rect(this.bbox[1],this.bbox[0],this.bbox[3]-this.bbox[1],this.bbox[2]-this.bbox[0])
101 
102 		// draw label if we're zoomed in enough'
103 		if (Cartagen.zoom_level > 0.3) {
104 			Cartagen.queue_label(this.label, this.x, this.y)
105 		}
106 	    canvas.restore()
107 	}
