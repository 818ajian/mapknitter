 14 // Wrapped native canvas methods in shorter, simpler method names:
 15 
 16 /**
 17  * Initializes the $C namespace
 18  */
 19 function canvas_init(){
 20 	/**
 21 	 * @namespace
 22 	 * Cavas functions, wapped into shorter, simpler names.
 23 	 * @see <a href="https://developer.mozilla.org/en/Canvas_tutorial/Drawing_shapes>MDC Docs</a>
 24 	 */
 25 	$C = {
 26 		/**
 27 		 * Clears the canvas
 28 		 * @function
 29 		 */
 30 		clear: function(){
 31 			canvas.clearRect(0, 0, width, height)
 32 		},
 33 		
 34 		/**
 35 		 * Sets canvas.fillStyle
 36 		 * @param {String} color Color to use for future fill operations
 37 		 */
 38 		fill_style: function(color) {
 39 			canvas.fillStyle = color
 40 		},
 41 		
 42 		/**
 43 		 * Alias of canvas.translate
 44 		 * @param {Number} x Number of pixels to tranlate in the x direction
 45 		 * @param {Number} y Number of pixels to tranlate in the y direction
 46 		 * @function
 47 		 */
 48 		translate: function(x,y) {
 49 			canvas.translate(x,y)
 50 		},
 51 		
 52 		/**
 53 		 * Alias of canvas.scale
 54 		 * @param {Number} x Number of pixels to stretch/shring in the x direction
 55 		 * @param {Number} y Number of pixels to stretch/shring in the y direction
 56 		 * @function
 57 		 */
 58 		scale: function(x,y) {
 59 			canvas.scale(x,y)
 60 		},
 61 		
 62 		/**
 63 		 * Alias of canvas.rotate
 64 		 * @param {Number} rotation Amount, in radians, to rotate
 65 		 * @function
 66 		 */
 67 		rotate: function(rotation){
 68 			canvas.rotate
 69 		},
 70 		
 71 		/**
 72 		 * Alias of canvas.fillRect (filled rectangle)
 73 		 * @param {Number} x X-coord of the top-left corner
 74 		 * @param {Number} y Y-coord of the top-left corner
 75 		 * @param {Number} w Width of the rectangle
 76 		 * @param {Number} h Height of the rectangle
 77 		 * @function
 78 		 */
 79 		rect: function(x, y, w, h){
 80 			canvas.fillRect(x, y, w, h)
 81 		},
 82 		
 83 		/**
 84 		 * Alias of canvas.strokeRect (unfilled rectangle
 85 		 * @param {Number} x X-coord of the top-left corner
 86 		 * @param {Number} y Y-coord of the top-left corner
 87 		 * @param {Number} w Width of the rectangle
 88 		 * @param {Number} h Height of the rectangle
 89 		 * @function
 90 		 */
 91 		stroke_rect: function(x, y, w, h){
 92 			canvas.strokeRect(x, y, w, h)
 93 		},
 94 		
 95 		/**
 96 		 * Alias of canvas.strokeStyle
 97 		 * @param {String} color Color to use for future stroke operations
 98 		 * @function
 99 		 */
100 		stroke_style: function(color) {
101 			canvas.strokeStyle = color
102 		},
103 		
104 		/**
105 		 * Sets canvas.lineWidth
106 		 * @param {Number} lineWidth New width, in pixels, to use for stroke operations
107 		 * @function
108 		 */
109 		line_width: function(lineWidth){
110 			if (parseInt(lineWidth) == 0) 
111 				canvas.lineWidth = 0.0000000001
112 			else 
113 				canvas.lineWidth = lineWidth
114 		},
115 		
116 		/**
117 		 * Alias of canvas.beginPath
118 		 * @function
119 		 */
120 		begin_path: function(){
121 			canvas.beginPath()
122 		},
123 		
124 		/**
125 		 * Alias of canvas.moveTo
126 		 * @param {Number} x X-coord of location to move to
127 		 * @param {Number} y Y-coord of location to move to
128 		 * @function
129 		 */
130 		move_to: function(x, y){
131 			canvas.moveTo(x, y)
132 		},
133 		
134 		/**
135 		 * Alias of canvas.lineTo
136 		 * @param {Number} x X-coord of location to draw line to
137 		 * @param {Number} y Y-coord of location to draw line to
138 		 * @function
139 		 */
140 		line_to: function(x, y){
141 			canvas.lineTo(x, y)
142 		},
143 		
144 		/**
145 		 * Draws a quadratic curve
146 		 * @param {Number} cp_x X-coord of control point
147 		 * @param {Number} cp_y Y-coord of control point
148 		 * @param {Number} x    X-coord of point to draw to
149 		 * @param {Number} y    Y-coord of point to draw to
150 		 * @see <a href="https://developer.mozilla.org/en/Canvas_tutorial/Drawing_shapes#Bezier_and_quadratic_curves">MDC Docs</a>
151 		 * @function
152 		 */
153 		quadratic_curve_to: function(cp_x, cp_y, x, y){
154 			canvas.quadraticCurveTo(cp_x, cp_y, x, y)
155 		},
156 		
157 		/**
158 		 * Draws a stroke along the current path.
159 		 * @function
160 		 */
161 		stroke: function(){
162 			canvas.stroke()
163 		},
164 		
165 		/**
166 		 * Closes the current path, then fills it.
167 		 * @function
168 		 */
169 		fill: function(){
170 			canvas.fill()
171 		},
172 		
173 		/**
174 		 * Draws an arc
175 		 * @param {Number} x                   X-coord of circle's center
176 		 * @param {Number} y                   Y-coord of circle's center
177 		 * @param {Number} radius              Radius of circle
178 		 * @param {Number} startAngle          Angle, in radians, from the +x axis to start the arc from
179 		 * @param {Number} endAngle            Angle, in radians, from the +x axis to end the arc at
180 		 * @param {Boolean} [counterclockwise] If true, arc is drawn counterclockwise. Else, it is drawn clockwise
181 		 * @function
182 		 */
183 		arc: function(x, y, radius, startAngle, endAngle, counterclockwise){
184 			canvas.arc(x, y, radius, startAngle, endAngle, counterclockwise)
185 		},
186 		
187 		/**
188 		 * Draws text on the canvas
189 		 * @param {String} font Font to use
190 		 * @param {Number} size Size, in pts, of text
191 		 * @param {Number} x    X-coord to start drawing at
192 		 * @param {Number} y    Y-coord to start drawing at
193 		 * @param {String} text Text to draw
194 		 * @function
195 		 */
196 		draw_text: function(font, size, x, y, text){
197 			canvas.drawText(font, size, x, y, text)
198 		},
199 		
200 		/**
201 		 * Draws text on canvas, with location specified as a center point
202 		 * @param {String} font Font to use
203 		 * @param {Number} size Size, in pts, of text
204 		 * @param {Number} x    X-coord to center text on
205 		 * @param {Number} y    Y-coord to center text on
206 		 * @param {String} text Text to draw
207 		 * @function
208 		 */
209 		draw_text_center: function(font, size, x, y, text){
210 			canvas.drawTextCenter(font, size, x, y, text)
211 		}
212 	}
213 }
