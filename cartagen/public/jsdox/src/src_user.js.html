181 					n.radius = User.node_radius
182 					n.fillStyle = node.color
183 					n.user = node.author
184 					n.lat = node.lat
185 					n.lon = node.lon
186 					// BAD!!! Why do we need a *-1????
187 					n.x = -1*Projection.lon_to_x(n.lon)
